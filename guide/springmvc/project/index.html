# 数据访问层 - 在entity目录中创建实体类User

package com.tms.three.layer.framework.demo.entity;

import java.io.Serializable;

/**
 * User类
 * @author liujia
 */
public class User implements Serializable {
    
    private String Id;
    private String userName;
    private String password;

    public String getId() {
        return Id;
    }

    public void setId(String id) {
        Id = id;
    }

    public String getUserName() {
        return userName;
    }

    public void setUserName(String userName) {
        this.userName = userName;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }

    @Override
    public String toString() {
        return "User{" +
                "Id='" + Id + '\'' +
                ", userName='" + userName + '\'' +
                ", password='" + password + '\'' +
                '}';
    }
}

# 数据访问层 - 在dao目录中创建UserDao接口

package com.tms.three.layer.framework.demo.dao;

import com.tms.three.layer.framework.demo.entity.User;

/**
 * UserDao 接口
 * @author liujia

public interface UserDao {

# 数据访问层 - 在dao目录中创建impl文件夹并实现UserDao接口

package com.tms.three.layer.framework.demo.dao.impl;

# 业务逻辑层 - 在service目录中创建UserService接口

package com.tms.three.layer.framework.demo.service;

import com.tms.three.layer.framework.demo.entity.User;

/**
 * userService 接口
 * @author liujia 
 */
public interface UserService {
    User login(String userName, String password);
}

# 业务逻辑层 - 在service目录中创建impl并实现UserService接口

package com.tms.three.layer.framework.demo.service.impl;

import com.tms.three.layer.framework.demo.dao.UserDao;
import com.tms.three.layer.framework.demo.dao.impl.UserDaoImpl;
import com.tms.three.layer.framework.demo.entity.User;
import com.tms.three.layer.framework.demo.service.UserService;

/**
 * 实现userService接口
 * @author liujia
 */
public class UserServiceImpl implements UserService {

    private UserDao userDao = new UserDaoImpl();
    public User login(String userName, String password) {
        System.out.println("userServices实现类UserServiceImpl,调用userDao接口");
        System.out.println("====进入数据访问层====");
        System.out.println("调用UserDao接口,获取UserDao实现类UserDaoImpl返回的查询结果");
        User user = userDao.login(userName, password);
        // 登录失败，user对象为空
        if (user == null){
            System.out.println("登录失败，user对象为空");
            return null;
        }
        // 登录成功，返回user对象
        else {
            System.out.println("登录成功，返回user对象");
            return user;
        }
    }
}

# 视图层 - 在controller目录下创建LoginController

package com.tms.three.layer.framework.demo.web.controller;

import com.tms.three.layer.framework.demo.entity.User;
import com.tms.three.layer.framework.demo.service.UserService;
import com.tms.three.layer.framework.demo.service.impl.UserServiceImpl;
import jakarta.servlet.ServletException;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;

import java.io.IOException;

public class LoginController extends HttpServlet {

    private UserService userService = new UserServiceImpl();
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        String userName = req.getParameter("userName");
        String password = req.getParameter("password");
        // 将访问login的请求交给userService 处理
        System.out.println("接收客户端登录请求后，调用LoginController");
        System.out.println("==== 进入业务逻辑层 ====");
        System.out.println("调用userService接口,获取userServices实现类UserServiceImpl返回的结果");
        User user = userService.login(userName, password);
        System.out.println("==== 返回视图层====");
        System.out.println("LoginController接收业务逻辑层userService返回的user对象：");

        // 登录失败，user对象为空，跳转到登录失败页面
        if(user == null){
            System.out.println("登录失败，user对象为空");
            req.getRequestDispatcher("/fail.jsp").forward(req, resp);
        }
        // 登录成功，返回user对象
        else {
            System.out.println("用户名："+user.getUserName());
            System.out.println("密码："+user.getPassword());
            req.getRequestDispatcher("/success.jsp").forward(req, resp);
        }
    }
}

# 视图层 - 在WEB-INF目录下创建web.xml

<?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns="http://xmlns.jcp.org/xml/ns/javaee"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://xmlns.jcp.org/xml/ns/javaee http://xmlns.jcp.org/xml/ns/javaee/web-app_4_0.xsd"
         version="4.0">
    <display-name>three-layer-framework-demo</display-name>
    <!--  servlet标签 给tomcat配置servlet程序-->
    <servlet>
        <!-- 3 servlet-name 标签给servlet起个别名-->
        <servlet-name>LoginController</servlet-name>
        <!-- 4 servlet-class是servlet全类名,告诉服务器LoginController的全类名 -->
        <servlet-class>com.tms.three.layer.framework.demo.web.controller.LoginController</servlet-class>
    </servlet>
    <!--    servlet-mapping 标签给servlet程序配置一个访问地址-->
    <servlet-mapping>
        <!-- 2 servlet-name 告诉服务器 请求登录地址对应LoginController-->
        <servlet-name>LoginController</servlet-name>
        <!-- 1 url-pattern 配置请求登录的访问地址-->
        <url-pattern>/login</url-pattern>
    </servlet-mapping>
</web-app>
